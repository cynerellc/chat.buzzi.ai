```typescript
interface Subscription {
  id: string;
  companyId: string;
  planId: string;
  status: 'active' | 'past_due' | 'canceled' | 'paused';
  billingCycle: 'monthly' | 'annual';
  currentPeriodStart: Date;
  currentPeriodEnd: Date;
  cancelAtPeriodEnd: boolean;
  canceledAt: Date | null;
  stripeSubscriptionId: string;
  stripeCustomerId: string;
}
```

### Usage
```typescript
interface Usage {
  companyId: string;
  period: {
    start: Date;
    end: Date;
  };
  messages: {
    used: number;
    limit: number;
    projected: number;
  };
  storage: {
    used: number;
    limit: number;
    breakdown: {
      knowledge: number;
      attachments: number;
    };
  };
  agents: {
    active: number;
    limit: number;
    usage: {
      agentId: string;
      name: string;
      messages: number;
    }[];
  };
  seats: {
    used: number;
    limit: number;
  };
}
```

```typescript
interface Invoice {
  id: string;

    description: string;
    quantity: number;
    unitAmount: number;
    amount: number;
  }[];
}
```
